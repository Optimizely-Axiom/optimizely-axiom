# Sidebar

Primary navigation menu for left side of the page that includes support for branding, links, etc.

## Documentation

### Usage

<Demo component={demos["sidebar/usage"]} />

### Anatomy

```tsx
import {
  Nav,
  NavAccountItem,
  NavBody,
  NavFooter,
  NavHeader,
  NavItem,
  Sidebar,
  SidebarToggle,
} from "@optiaxiom/react";

export default () => (
  <Sidebar>
    <Nav>
      <NavHeader>
        <NavItem />
      </NavHeader>

      <NavBody>
        <NavItem />
      </NavBody>

      <NavFooter>
        <SidebarToggle />
        <NavItem />
        <NavAccountItem />
      </NavFooter>
    </Nav>
  </Sidebar>
);
```
