### Anatomy

```tsx
import {
  Nav,
  NavAccountItem,
  NavBody,
  NavFooter,
  NavHeader,
  NavItem,
  Sidebar,
  SidebarToggle,
  SubNav,
} from "@optiaxiom/react";

export default () => (
  <Sidebar>
    <Nav>
      <NavHeader>
        <NavItem />
      </NavHeader>

      <NavBody>
        <NavItem />
      </NavBody>

      <NavFooter>
        <SidebarToggle />
        <NavItem />
        <NavAccountItem />
      </NavFooter>
    </Nav>

    <SubNav />
  </Sidebar>
);
```
