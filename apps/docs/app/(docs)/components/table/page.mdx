### Anatomy

```tsx
import {
  Table,
  TableBody,
  TableCell,
  TableHeader,
  TableHeaderCell,
  TableRow,
} from "@optiaxiom/react/unstable";

  <Table>
    <TableHeader>
      <TableRow>
        <TableHeaderCell />
      </TableRow>
    </TableHeader>

    <TableBody>
      <TableRow>
        <TableCell />
      </TableRow>
    </TableBody>
  </Table>
);
```
