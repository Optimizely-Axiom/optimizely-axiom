# Autocomplete

Text field with inline autocomplete to show suggestions while allowing freeform user input.

## Documentation

### Anatomy

```tsx
import {
  Autocomplete,
  AutocompleteContent,
  AutocompleteEmptyItem,
  AutocompleteItem,
  AutocompleteItemIndicator,
  AutocompleteList,
  AutocompleteTrigger,
} from "@optiaxiom/react";

export default () => (
  <Autocomplete>
    <AutocompleteTrigger />

    <AutocompleteContent>
      <AutocompleteList>
        <AutocompleteItem>
          <AutocompleteItemIndicator />
        </AutocompleteItem>
      </AutocompleteList>

      <AutocompleteEmptyItem />
    </AutocompleteContent>
  </Autocomplete>
);
```
