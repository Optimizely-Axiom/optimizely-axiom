### Anatomy

```tsx
import {
  Autocomplete,
  AutocompleteContent,
  AutocompleteEmpty,
  AutocompleteItem,
  AutocompleteRadioItem,
  AutocompleteTrigger,
} from "@optiaxiom/react";

export default () => (
  <Autocomplete>
    <AutocompleteTrigger />

    <AutocompleteContent>
      <AutocompleteItem />
      <AutocompleteRadioItem />
      <AutocompleteEmpty />
    </AutocompleteContent>
  </Autocomplete>
);
```
