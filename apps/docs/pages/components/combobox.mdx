# Combobox

Multi-purpose combobox widget to allow selection from a dynamic set of options.

## Documentation

### Usage

<Demo component={Usage} />

### Anatomy

```tsx
import {
  Combobox,
  ComboboxCheckboxItem,
  ComboboxContent,
  ComboboxEmpty,
  ComboboxFooter,
  ComboboxGroup,
  ComboboxInput,
  ComboboxItem,
  ComboboxLabel,
  ComboboxListbox,
  ComboboxRadioItem,
  ComboboxScrollArea,
  ComboboxSeparator,
  ComboboxTrigger,
  ComboboxValue,
} from "@optiaxiom/react";

export default () => (
  <Combobox>
    <ComboboxTrigger>
      <ComboboxValue />
    </ComboboxTrigger>

    <ComboboxContent>
      <ComboboxInput />

      <ComboboxListbox>
        <ComboboxScrollArea>
          {/**
           * Everything inside <ComboboxListbox /> except for:
           *  - <ComboboxFooter />
           *  - <ComboboxScrollArea />
           */}
        </ComboboxScrollArea>

        <ComboboxItem />
        <ComboboxCheckboxItem />
        <ComboboxRadioItem />

        <ComboboxGroup>
          <ComboboxLabel />
          <ComboboxItem />
          <ComboboxCheckboxItem />
          <ComboboxRadioItem />
        </ComboboxGroup>

        <ComboboxEmpty />
        <ComboboxFooter />
        <ComboboxSeparator />
      </ComboboxListbox>
    </ComboboxContent>
  </Combobox>
);
```
