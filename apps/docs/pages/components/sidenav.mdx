# Sidenav

Primary navigation menu for left side of the page that includes support for branding, links, etc.

## Documentation

### Usage

<Demo component={Usage} />

### Anatomy

```tsx
import {
  Sidenav,
  SidenavAccountItem,
  SidenavBody,
  SidenavFooter,
  SidenavHeader,
  SidenavItem,
  SidenavToggle,
} from "@optiaxiom/react";

export default () => (
  <Sidenav>
    <SidenavHeader>
      <SidenavItem />
    </SidenavHeader>

    <SidenavBody>
      <SidenavItem />
    </SidenavBody>

    <SidenavFooter>
      <SidenavToggle />
      <SidenavItem />
      <SidenavAccountItem />
    </SidenavBody>
  </Sidenav>
);
```
